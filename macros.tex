% Commands

\newcommand{\drawcard}[2]{\raisebox{-2 pt}{%
		\begin{tikzpicture}
			\draw (-0.2,-0.2) rectangle (0.2,0.2)
				(0,0) node {
					\scalerel*{ \hspace{0 mm}
						\def\svgwidth{10 pt}
						\import{cards/}{#1.pdf_tex}
					}{X\rule[0ex]{0pt}{1pt}}}
					#2;
		\end{tikzpicture}}%
	}

% Generic card template

\makeatletter

\newcommand{\C}[2][]{%
	%		
	\IfBeginWith{#2}{R}{\def\@color{\red}}{}%
	\IfBeginWith{#2}{Y}{\def\@color{\yellow}}{}%
	\IfBeginWith{#2}{G}{\def\@color{\green}}{}%
	\IfBeginWith{#2}{B}{\def\@color{\blue}}{}%
	\IfBeginWith{#2}{W}{\def\@color{\white}}{}%
	\IfBeginWith{#2}{M}{\def\@color{\rainbow}}{}%
	%
	\IfEndWith{#2}{0}{\def\@number{\zero}}{}%
	\IfEndWith{#2}{1}{\def\@number{\one}}{}%
	\IfEndWith{#2}{2}{\def\@number{\two}}{}%
	\IfEndWith{#2}{3}{\def\@number{\three}}{}%
	\IfEndWith{#2}{4}{\def\@number{\four}}{}%
	\IfEndWith{#2}{5}{\def\@number{\five}}{}%
	%
	\ifthenelse{\equal{#1}{n}}{\drawcard{#2}{\colorclue{\ZZ} \numberclue{\@number}}}{%
		\ifthenelse{\equal{#1}{c}}{\drawcard{#2}{\colorclue{\@color} \numberclue{\ZZ}}}{%
			\ifthenelse{\equal{#1}{cn}}{\drawcard{#2}{\colorclue{\@color} \numberclue{\@number}}}{%
				\drawcard{#2}{\colorclue{\ZZ} \numberclue{\ZZ}}}}}\!%
}
\makeatother
